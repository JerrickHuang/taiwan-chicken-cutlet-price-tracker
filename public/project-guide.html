五、SQLite 資料庫設計

資料表為 price_records，支援單一年份、期間資料、價格區間、來源與推估註記。

欄位用途
record_date使用者輸入日期或資料代表日期
item_name商品名稱，預設為雞排
period顯示用期間，例如 1993-1995
year_start / year_end支援年份滑桿與圖表計算
price_ntd代表價格
price_min_ntd / price_max_ntd價格區間
is_estimated0 為來源紀錄，1 為推估資料
source_ref / source_url來源代號與來源網址
Azure 部署時資料庫路徑會自動改成 /home/data/data.db，避免每次重新部署就清掉雲端資料。

1. 前端：使用 fetch 呼叫後端 API

檔案位置：public/app.js

async function fetchJson(url, options = {}) {
  const response = await fetch(url, options);
  const data = await response.json();

  if (!response.ok) {
    throw new Error(data.error || '資料讀取失敗。');
  }

  return data;
}

async function loadData() {
  const query = els.searchInput.value.trim();
  const url = query ? `/api/prices?q=${encodeURIComponent(query)}` : '/api/prices';
  const [{ records }, { summary }] = await Promise.all([
    fetchJson(url),
    fetchJson('/api/prices/summary')
  ]);

  state.records = records;
  state.summary = summary;
  renderAll();
}

說明：前端透過 fetchJson() 呼叫 Express API，取得價格資料與摘要資料後，更新表格、摘要卡、年份滑桿與折線圖。

2. 後端：Express.js 查詢與新增 API

檔案位置：server.js

app.get('/api/prices', async (req, res, next) => {
  try {
    const { where, params } = buildPriceQuery(req.query);
    const records = await all(db, `
      SELECT *
      FROM price_records
      ${where}
      ORDER BY COALESCE(year_start, year_end), id
    `, params);

    res.json({ records });
  } catch (error) {
    next(error);
  }
});

app.post('/api/prices', async (req, res, next) => {
  try {
    const record = await insertPriceRecord(db, req.body);
    res.status(201).json({ record });
  } catch (error) {
    next(error);
  }
});

說明：GET /api/prices 負責查詢 SQLite 資料，支援搜尋與年份篩選；POST /api/prices 接收前端新增資料並寫入資料庫。

3. 資料庫：建立 SQLite 資料表並新增資料

檔案位置：src/database.js

async function createSchema(db) {
  await run(db, `
    CREATE TABLE IF NOT EXISTS price_records (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      record_date TEXT NOT NULL,
      item_name TEXT NOT NULL,
      period TEXT NOT NULL,
      year_start INTEGER,
      year_end INTEGER,
      price_ntd INTEGER NOT NULL,
      price_min_ntd INTEGER,
      price_max_ntd INTEGER,
      is_estimated INTEGER NOT NULL DEFAULT 0,
      source_ref TEXT NOT NULL DEFAULT 'USER',
      source_url TEXT DEFAULT '',
      note TEXT DEFAULT '',
      created_at TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
    )
  `);
}

說明：createSchema() 會建立 price_records 資料表，儲存日期、商品名稱、價格、年份區間、來源與備註等欄位，讓新增資料重新整理後仍能保留。

4. 加分功能：爬蟲擷取可能價格

檔案位置：server.js

function extractPriceCandidates(text) {
  const candidates = new Map();
  const patterns = [
    /雞排[^。！？\n]{0,80}?(?:NT\$|NTD|新台幣|台幣)?\s*(\d{2,3})\s*元/g,
    /(?:NT\$|NTD|新台幣|台幣)\s*(\d{2,3})\s*元?[^。！？\n]{0,80}?雞排/g,
    /雞排[^。！？\n]{0,80}?(\d{2,3})\s*(?:元|塊)/g
  ];

  for (const pattern of patterns) {
    for (const match of text.matchAll(pattern)) {
      const price = Number(match[1]);
      if (Number.isInteger(price) && price >= 20 && price <= 250) {
        candidates.set(price, {
          price_ntd: price,
          snippet: match[0].replace(/\s+/g, ' ').trim().slice(0, 120)
        });
      }
    }
  }

  return [...candidates.values()]
    .sort((a, b) => b.price_ntd - a.price_ntd)
    .slice(0, 5);
}

說明：爬蟲 API 會讀取公開來源網頁，清理 HTML 後用正規表示式尋找「雞排」附近的價格文字，回傳建議價格與最多 5 筆候選片段，供使用者人工確認。
